**修复**:
```rust
// 修复前 - 可能panic

// 修复后 - 安全处理
let api_key = api_key.ok_or_else(|| anyhow::anyhow!("API key not available"))?;
let selector = Selector::parse("article")
    .map_err(|e| anyhow::anyhow!("Failed to parse selector: {}", e))?;
```
